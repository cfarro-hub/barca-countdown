// ⏳ COUNTDOWN TIMER const targetDate = new Date("2026-06-01T00:00:00").getTime(); function updateCountdown() { const now = new Date().getTime(); const diff = targetDate - now; const days = Math.floor(diff / (1000 * 60 * 60 * 24)); const hours = Math.floor((diff % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60)); const minutes = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60)); const seconds = Math.floor((diff % (1000 * 60)) / 1000); document.getElementById("countdown").innerHTML = `${days}d ${hours}h ${minutes}m ${seconds}s`; } setInterval(updateCountdown, 1000); updateCountdown();

  <script>
    const targetDate = new Date("2026-06-01T12:00:00");
    let celebrated = false;

    function playSound() {
      const audio = document.getElementById("crowd");
      audio.play();
    }

    function flip(id, value) {
      const el = document.getElementById(id);
      el.classList.remove("flip");
      void el.offsetWidth;
      el.innerText = value;
      el.classList.add("flip");
    }

    function updateCountdown() {
      const now = new Date();
      const diff = targetDate - now;

      if (diff <= 0) {
        document.querySelector(".countdown").innerHTML = "<h2>They’re Here! 🎉</h2>";

        if (!celebrated) {
          launchConfetti();
          celebrated = true;
        }
        return;
      }

      const days = Math.floor(diff / (1000 * 60 * 60 * 24));
      const hours = Math.floor((diff / (1000 * 60 * 60)) % 24);
      const minutes = Math.floor((diff / 1000 / 60) % 60);
      const seconds = Math.floor((diff / 1000) % 60);

      flip("days", days);
      flip("hours", hours);
      flip("minutes", minutes);
      flip("seconds", seconds);
    }

    function launchConfetti() {
      const duration = 5000;
      const end = Date.now() + duration;
      const colors = ["#A50044", "#004D98", "#FFED02"];

      (function frame() {
        confetti({ particleCount: 5, angle: 60, spread: 70, origin: { x: 0 }, colors });
        confetti({ particleCount: 5, angle: 120, spread: 70, origin: { x: 1 }, colors });
        if (Date.now() < end) requestAnimationFrame(frame);
      })();
    }

    setInterval(updateCountdown, 1000);
    updateCountdown();
  </script>
